# 推荐: 使用 Annotated 自定义类型
class StrippedStr(str):
    @classmethod
    def __get_validators__(cls):
        yield cls.validate
    @classmethod
    def validate(cls, v):
        return str(v).strip() if v else ""

# 使用方式
